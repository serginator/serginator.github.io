193     *
194     * @augments Array
195     * @param m {int} Number of columns of the matrix
196     * @param n {int} Number of raws of the matrix
197     * @param initial {int} The init value of the elements
198     * @return {mat} Returns the matrix
199     */
200     Array.matrix = function(m, n, initial){
201         var a, i, j, mat = [];
202         for(i = 0; i < m; i++){
203             a = [];
204             for(j = 0; j < n; j++){
205                 a[j] = initial;

208         }

413     /**
414     * Construct a new Snake object.
415     *
416     * @class This is the class for the Snake
417     * @constructor
418     * @param board {Board} The board
419     * @param fruit {Fruit} The fruit
420     * @return {this} Returns the Snake
421     */
422     function Snake(board, fruit){
423         /** @private */ var lives = 3;
424         /** @private */ var posX = 6;
425         /** @private */ var posY = 13;
426         /** @private */ var auxX = 0;
427         /** @private */ var auxY = 0;
428         /** @private */ var tailX = 6;
429         /** @private */ var tailY = 13;
430         /** @private */ var arrayMoves = ['right']; //To save movements for tail moving later
431         /** @private */ var currentMove = arrayMoves[arrayMoves.length - 1];
432         /** @private */ var nextMove = 'right'; //To anticipate collisions        
433         /** @private */ var increaseTail = 3; //It will be 3 every time we eat a fruit
434         
435         board.setValue(posX, posY, 1); //To initialize the Snake on the board
436         
437         /**
438         * Get currentMove
439         *
440         * @return {String} Return the value
441         */
442 		this.getCurrentMove = function(){
443 			return currentMove;
444 		};
445 		
446 		/**
447         * Set score
448         *
449         * @param move {String} The next move to set
450         * @return {this} To be able to chain methods
451         */
452 		this.setNextMove = function(move){
453 			nextMove = move;
454 			return this;
455 		};
456 		
457 		/**
458         * Get lives
459         *
460         * @return {int} Lives of the snake
461         */
462 		this.getLives = function(){
463 		    return lives;
464 		};
465 		
466 		/**
467         * Check the key pressed for the snake movement. As I have to
468         * use snake, I couldn't use it outside Snake "class" so here is it.
469         *
470         * @return {this} To be able to chain methods.
471         */
472 		this.checkKeys = function(){
473 		    if(Key.isDown(Key.UP)){
474                 if(currentMove !== 'down'){
475                     nextMove = 'up';
476                 }
477             }
478             if(Key.isDown(Key.DOWN)){
479                 if(currentMove !== 'up'){
480                     nextMove = 'down';
481                 }
482             }
483             if(Key.isDown(Key.LEFT)){
484                 if(currentMove !== 'right'){
485                     nextMove = 'left';
486                 }
487             }
488             if(Key.isDown(Key.RIGHT)){
489                 if(currentMove !== 'left'){
490                     nextMove = 'right';
491                 }
492             }
493             return this;
494 		};
495 		
496 		
497         /**
498         * Check if the next movement can be done.
499         * Catch the last movement on the array. Copy the coordinates on 
500         * auxiliary ones. Change the auxiliary coordinates.
501         * If that position has a 1 on the board check if it's a fruit or a wall.
502         * If it's a fruit or the position is 0, make the movement.
503         *
504         * @param board {Board} The board of the game to get values
505         * @param fruit {Fruit} The fruit, to check coordenates
506         * @return {Boolean} If the movement can be done, return true.
507         */
508         this.checkMove = function(board, fruit){
509             /** @private */ var move = arrayMoves[arrayMoves.length - 1];
510             auxX = posX;
511             auxY = posY;
512             if(move === 'up'){
513                 auxY--;
514                 currentMove = 'up';
515             }
516             else if(move === 'down')
517             {
518                 auxY++;
519                 currentMove = 'down';
520             }
521             else if(move === 'left'){
522                 auxX--;
523                 currentMove = 'left';
524             }
525             else if(move === 'right'){
526                 auxX++;
527                 currentMove = 'right';
528             }
529             if(board.getValue(auxX, auxY) === 1){ //A fruit or a wall
530                 if((auxX === fruit.getPosX()) && (auxY === fruit.getPosY())){ //A fruit
531                     this.eatFruit(board, fruit);
532                     return true;
533                 }
534                 else{ //A wall
535                     return false;
536                 }
537             }
538             else{ //A free move
539                 return true;
540             }
541         };
542         
543         /**
544         * Move the Snake.
545         * Check if the movement can be done.
546         * If true, copy the auxiliary coordinates to the real ones
547         * and push the nextMove to the arrayMoves.
548         *
549         * @param board {Board} The board of the game
550         * @param fruit {Fruit} The fruit
551         * @return {Boolean} To check if we have to move the Tail or not
552         */
553         this.moveSnake = function(board, fruit){
554             if(this.checkMove(board, fruit)){
555                 posX = auxX;
556                 posY = auxY;
557                 board.setValue(posX, posY, 1);
558                 arrayMoves.push(nextMove);
559                 return true;
560             }
561             else{
562                 lives--;
563                 if(lives === 0){
564                     gameOver = true;
565                     lives = 3;
566                     yDown = false;
567                 }
568                 board.initBoard();
569                 posX = 6;
570                 posY = 13;
571                 tailX = 6;
572                 tailY = 13;
573                 arrayMoves = ['right']; //To save movements for tail moving later
574                 currentMove = arrayMoves[arrayMoves.length - 1];
575                 nextMove = 'right'; //To anticipate collisions        
576                 increaseTail = 3; //It will be 3 every time we eat a fruit
577                 board.setValue(posX, posY, 1);
578                 //resetea fruta
579                 fruit.putNewFruit(board);
580                 return false;
581             }
582         };
583         
584         /**
585         * Move the tail of the Snake.
586         * If the tail is increasing, decrease it by 1.
587         * Else set the value of the tail on the array at 0.
588         * Catch the first movement on the arrayMoves.
589         * Change the coordinate.
590         *
591         * @param board {Board} The board to set values
592         * @return {this} To be able to chain functions
593         */
594         this.moveTail = function(board){
595             if(increaseTail === 0){
596                 board.setValue(tailX, tailY, 0);
597                 /** @private */ var move = arrayMoves.shift();
598                 if(move === 'up'){
599 					tailY--;
600 				}
601                 else if(move === 'down'){
602 					tailY++;
603 				}
604                 else if(move === 'left'){
605 					tailX--;
606 				}
607                 else if(move === 'right'){
608 					tailX++;
609 				}
610             }
611             else{
612                 increaseTail--;
613             }
614             return this;
615         };
616         
617         /**
618         * Eat the Fruit.
619         * Set increaseTail value to 3. Increase the board.score.
620         * Randomize the coordinates of the Fruit.
621         *
622         * @param board {Board} The board to set the score
623         * @param fruit {Fruit} The fruit to put a new one
624         * @return {this} To be able to chain functions
625         */
626         this.eatFruit = function(board, fruit){
627             increaseTail = 3;
628             board.setScore(board.getScore() + fruit.getScore());
629             fruit.putNewFruit(board);
630             return this;
631         };
632         return this;
633     }
